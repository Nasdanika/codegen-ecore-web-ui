001package org.nasdanika.codegen.ecore.web.ui;
002
003import java.util.Collection;
004import java.util.Collections;
005
006import org.nasdanika.codegen.GeneratorController;
007import org.nasdanika.codegen.java.ImportManager;
008import org.nasdanika.codegen.java.Interface;
009import org.nasdanika.config.Context;
010import org.nasdanika.config.MutableContext;
011
012/**
013 * 
014 * @author Pavel
015 *
016 */
017public class RendererBaseController implements GeneratorController<String, Interface> {
018
019        @Override
020        public Collection<Context> iterate(Context context, Interface generator) throws Exception {
021                MutableContext mc = context.createSubContext();
022                ImportManager importManager = mc.get(ImportManager.class);
023                mc.set("renderer-base-extends-imported", importManager.addImport((String) mc.get(RoutesPackageController.RENDERER_BASE_EXTENDS)));
024                return Collections.singleton(mc);
025        }
026
027}
